function checkall_sector(obj_id, status, main_id = "", from = ""){
    if(main_id != ""){
        $("#" + main_id).prop("checked", status);

        if(from == "check_box"){
            check_sector();
        }
    }
    
    $("div#" + obj_id).find("input").each(function(){
        $("#" + this.id).prop("checked", status);
    });
}

function check_sector(){

    var check = false;
    $("div#pref_area").find("input[name='sv_sector[]']").each(function(){
        if($("#" + this.id).is(':checked')){
            check = true;
        } else {
            check = false;
            return false;
        }
    });
    
    if(check == true){
        $("#sv_all").prop("checked", true);
    } else {
        $("#sv_all").prop("checked", false);
    }
}

function check_pref(obj_id, status, now_obj){

    var check = false;
    $("div#" + obj_id).find("input[name='sv_pref_cd[]']").each(function(){
        if($("#" + this.id).is(':checked')){
            check = true;
        } else {
            check = false;
            return false;
        }
    });

    if(check == true){
        $("#" + now_obj).prop("checked", true);
    } else {
        $("#" + now_obj).prop("checked", false);
    }

    check_sector();
}

2020/03/06 save relation table in one save with cakePHP model

table example:

table_a (id, name, url)
table_b (id, name, url, table_a_id)

* table_a hasMany table_b
* table_b belongsTo table_a
* foreignKey is table_a_id

step 1: create form inputs
ref: https://book.cakephp.org/3/en/views/helpers/form.html#creating-inputs-for-associated-data

// input for table_a
echo $this->Form->control('name');

// input for table_b
// format: table_name.array_index.field_name
// array_index start at 0
echo $this->Form->control('table_b.0.name');
// warning: if not table_name.array_index.primary_field_name will add new alway

exmple:
echo $this->Form->control('name');

2020/06/15 ยกเลิก git ที่ commit ไปแล้ว แต่ยังไม่ได้ push ไปที่ git

git reset --soft HEAD~1
หลังใช้คำสั่งนี้แล้วให้เก็บไฟล์ที่ commit แล้วไว้ใน stage ก่อน

git reset --hard HEAD~1
หลังใช้คำสั่งนี้แล้วจะลบไฟล์ที่ commit แล้วออกไป

Reference : https://www.git-tower.com/learn/git/faq/undo-last-commit
